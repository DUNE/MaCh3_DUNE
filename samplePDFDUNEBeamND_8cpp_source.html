332// Set the covariance matrix for this class
333void samplePDFDUNEBeamND::setNDCovMatrix() {
334 if (NDCovMatrix == NULL) {
335 std::cerr << "Could not find ND Detector covariance matrix you provided to setCovMatrix" << std::endl;
336 std::cerr << __FILE__ << ":" << __LINE__ << std::endl;
337 throw;
338 }
339
340 int nXBins = static_cast<int>(XBinEdges.size()-1);
341 int nYBins = static_cast<int>(YBinEdges.size()-1);
342 int covSize = nXBins*nYBins;
343
344 if (covSize != NDCovMatrix->GetNrows()) {
345 std::cerr << "Sample dimensions do not match ND Detector Covariance!" << std::endl;
346 std::cerr << "Sample XBins * YBins = " << covSize << std::endl;
347 std::cerr << "ND Detector Covariance = " << NDCovMatrix->GetNrows() << std::endl;
348 std::cerr << __FILE__ << ":" << __LINE__ << std::endl;
349 throw;
350 }
351
352 std::vector<double> FlatCV;
353 int iter = 0;
354
355 // 2D -> 1D Array
356 for (int xBin = 0; xBin < nXBins; xBin++)
357 {
358 for (int yBin = 0; yBin < nYBins; yBin++)
359 {
360 double CV = samplePDFFD_data[yBin][xBin];
361 FlatCV.push_back(CV);
362
363 if(CV>0) (*NDCovMatrix)(iter,iter) += 1/CV;
364
365 iter++;
366 }
367 }
368
369 NDInvertCovMatrix = new double*[covSize]();
370 // Set the defaults to true
371 for(int i = 0; i < covSize; i++)
372 {
373 NDInvertCovMatrix[i] = new double[covSize]();
374 for (int j = 0; j < covSize; j++)
375 {
376 NDInvertCovMatrix[i][j] = 0.;
377 }
378 }
379
380 TMatrixD* NDInvCovMatrix=static_cast<TMatrixD*>(NDCovMatrix->Clone());
381 NDInvCovMatrix->Invert();
382
383
384 //Scale back to inverse absolute cov and use standard double
385 for (int i = 0; i < covSize; i++) {
386 for (int j = 0; j < covSize; ++j) {
387 const double f = FlatCV[i] * FlatCV[j];
388 if(f != 0) NDInvertCovMatrix[i][j] = (*NDInvCovMatrix)(i,j)/f;
389 else NDInvertCovMatrix[i][j] = 0.;
390 }
391 }
392
393}

395//New likelihood calculation for ND samples using detector covariance matrix
396double samplePDFDUNEBeamND::GetLikelihood()
397{
398 if (samplePDFFD_data == NULL) {
399 std::cerr << "data sample is empty!" << std::endl;
400 return -1;
401 }
402
403 //if (NDInvertCovMatrix == NULL) {
404 //setNDCovMatrix();
405 //}
406
407 if (!isNDCovSet) {
408 setNDCovMatrix();
409 isNDCovSet = true;
410 }
411
412 int nXBins = static_cast<int>(XBinEdges.size()-1);
413 int nYBins = static_cast<int>(YBinEdges.size()-1);
414
415 int covSize = nXBins*nYBins;
416
417 std::vector<double> FlatData;
418 std::vector<double> FlatMCPred;
419
420 //2D -> 1D
421 for (int xBin = 0; xBin < nXBins; xBin++)
422 {
423 for (int yBin = 0; yBin < nYBins; yBin++)
424 {
425 double MCPred = samplePDFFD_array[yBin][xBin];
426 FlatMCPred.push_back(MCPred);
427
428 double DataVal = samplePDFFD_data[yBin][xBin];
429 FlatData.push_back(DataVal);
430 }
431 }
432
433
434 double negLogL = 0.;
435#ifdef MULTITHREAD
436#pragma omp parallel for reduction(+:negLogL)
437#endif
438
439 for (int i = 0; i < covSize; i++) {
440 for (int j = 0; j <= i; ++j) {
441 //KS: Since matrix is symetric we can calcaute non daigonal elements only once and multiply by 2, can bring up to factor speed decrease.
442 int scale = 1;
443 if(i != j) scale = 2;
444 negLogL += scale * 0.5*(FlatData[i] - FlatMCPred[i])*(FlatData[j] - FlatMCPred[j])*NDInvertCovMatrix[i][j];
445 }
446 }
447
448 return negLogL;
449}
