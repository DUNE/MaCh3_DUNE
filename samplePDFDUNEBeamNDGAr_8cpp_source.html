317 theta_start = atan2(_MCPStartY->at(i_anapart) - centre_circle_y, _MCPStartZ->at(i_anapart) - centre_circle_z);
318
319 //Ensure they are in the range [0,2pi)
320 if (theta_1 < 0) theta_1 += 2*M_PI;
321 if (theta_2 < 0) theta_2 += 2*M_PI;
322 if (theta_start < 0) theta_start += 2*M_PI;
323
324 //Lorentz force law, if positively charged, counter clockwise, if negative charged, clockwise
325 if(!positivecharged){
326 theta_diff_1 = (theta_1 < theta_start) ? (theta_start - theta_1) : (2*M_PI - (theta_1 - theta_start));
327 theta_diff_2 = (theta_2 < theta_start) ? (theta_start - theta_2) : (2*M_PI - (theta_2 - theta_start));
328 }
329 else if(positivecharged){
330 theta_diff_1 = (theta_1 > theta_start) ? (theta_1 - theta_start) : (2*M_PI - (theta_start - theta_1));
331 theta_diff_2 = (theta_2 > theta_start) ? (theta_2 - theta_start) : (2*M_PI - (theta_start - theta_2));
332 }
333
334 //JM Why is the second check necessary?
335 if(theta_diff_1<theta_diff_2 && (rad_curvature*100*theta_diff_1 > (TPCInstrumentedRadius-start_radius))){

337 }
338 else {
339 theta_chosen = theta_diff_2;
340 }
341
342 L_yz = rad_curvature*100*theta_chosen;
343 L_yz_chord = std::abs(2*rad_curvature*100*sin(theta_chosen/2));
344 nturns = std::abs(length_track_x/100)/pitch;
345
346 //If track ends before it would intersect, adjust
347 if(std::abs(L_yz*tan_theta) > std::abs(length_track_x)){
348 double theta_intersect = fmod(nturns, 1)*2*M_PI; //JM why do we need the fmod? nturns<1 in this case surely?
349
350 L_yz = rad_curvature*100*theta_intersect;
351 L_yz_chord = std::abs(2*rad_curvature*100*sin(theta_intersect/2));
352 }
353 else {
354 length_track_x = std::abs(L_yz*tan_theta);
355 }
356 }
357 else { //Radius of curvature small enough to spiral within TPC
358 nturns = std::abs(length_track_x/100)/pitch;
359 double theta_intersect = nturns*2*M_PI;
360 L_yz = theta_intersect*rad_curvature*100;
361 L_yz_chord = L_yz;
362 //if (nturns>=1) {theta_intersect = 2*M_PI;} //JM Why cap at 2pi?
363 //else {theta_intersect = nturns*2*M_PI;}
364 //L_yz_chord = std::abs(2*rad_curvature*100*sin(theta_intersect/2));
365 //L_yz = rad_curvature*100*theta_intersect; //JM this was previously uninitialised in this case
366 }
367
368 double N_hits = FindNHits(pixel_spacing_cm, centre_circle_y, centre_circle_z, rad_curvature);
369 double p_mag = sr->mc.nu[0].prim[i_truepart].p.Mag();
370 double bg = 0;
371 double gamma = 0;
372 double beta = CalcBeta(p_mag, bg, gamma, pdgmass);
373
374 //JM removed avg_betapT calc (seems to be unused)
375
376 double sigmax = (drift_velocity/100)*(1/(adc_sampling_frequency));
377 double sigmax_frac = sigmax/(std::abs(length_track_x)/100);
378 double sigmayz = (spatial_resolution/(1000)); //needs to be in m
379 double momres_yz = transverse_mom*(std::sqrt(720/(N_hits+4)) * (sigmayz*transverse_mom/(0.3*B_field*(L_yz_chord/100)*(L_yz_chord/100)))
380 * std::sqrt(1-(1/21)*(L_yz_chord/(rad_curvature*100))*(L_yz_chord/(rad_curvature*100))));
381 double momres_ms = transverse_mom*(0.016/(0.3*B_field*(L_yz/100)*cos(theta_xT)*beta))*std::sqrt(L_yz/X0);
382 double momres_tottransverse = std::sqrt(momres_yz*momres_yz + momres_ms*momres_ms)/transverse_mom;
383 double sigma_theta = (cos(theta_xT)*cos(theta_xT) * (pitch/(2*M_PI*rad_curvature)) *
384 std::sqrt(sigmax_frac*sigmax_frac + momres_tottransverse*momres_tottransverse));
385 double momres_frac = std::sqrt(momres_tottransverse*momres_tottransverse + (sigma_theta*tan_theta)*(sigma_theta*tan_theta));
386
387 duneobj->particle_nhits->back() = N_hits;
388 duneobj->particle_nturns->back() = nturns;
389 duneobj->particle_momresms->back() = momres_ms;
390 duneobj->particle_momrestransfrac->back() = momres_tottransverse/momres_frac;
391 duneobj->particle_momrestrans->back() = momres_tottransverse;
392
393 if(momres_frac > momentum_resolution_threshold){

395 }
396 }
397 else{ //Neutral particles: accept pi0 and gamma with some efficiency
398 if((std::abs(pdgcaf)) == 111){
399 double pi0recoprob = dist(gen);
400 if(pi0recoprob>pi0_reco_efficiency){

403 }
404 else if((std::abs(pdgcaf)) == 22){
405 double gammarecoprob = dist(gen);
406 if(gammarecoprob>gamma_reco_efficiency){
407 isAccepted = false;
408 }
409 }
410 else{
411 isAccepted = false;
412 }
413 }
414 }
415 else if (stops_in_ecal){
416 double energydepsum = 0;
417 for(unsigned int i_ecaldep=0; i_ecaldep<_SimHitEnergy->size(); i_ecaldep++){
418 if(_SimHitTrkID->at(i_ecaldep) == _MCPTrkID->at(i_anapart)){
419 energydepsum = energydepsum + _SimHitEnergy->at(i_ecaldep);
420 }
421 }
422 duneobj->particle_ecaldepositfraction->back() = energydepsum/(sr->mc.nu[0].prim[i_truepart].p.E);
423 }
424 if (isAccepted == false) {duneobj->particle_isaccepted->back() = false;}
425 break;
426 }
427 if(nummatched != 1){
428 MACH3LOG_INFO("Found {} matching particles in anatree", nummatched);
429 MACH3LOG_INFO("PDG: {}, momentum: ({},{},{})",pdgcaf,sr->mc.nu[0].prim[i_truepart].p.px,sr->mc.nu[0].prim[i_truepart].p.py,sr->mc.nu[0].prim[i_truepart].p.pz);
430 }
431 return isAccepted;
